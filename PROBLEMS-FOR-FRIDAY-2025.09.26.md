### **JavaScript**

```javascript
// Yarn trail effect
const yarnTrail = [];
const trailLength = 10; // number of dots in trail

// Create trail dots
for (let i = 0; i < trailLength; i++) {
  const dot = document.createElement('div');
  dot.classList.add('yarn-dot');
  document.body.appendChild(dot);
  yarnTrail.push(dot);
}

let mouseX = 0;
let mouseY = 0;

document.addEventListener('mousemove', (e) => {
  mouseX = e.clientX;
  mouseY = e.clientY;
});

// Animate trail
function animateTrail() {
  let x = mouseX;
  let y = mouseY;

  yarnTrail.forEach((dot, index) => {
    dot.style.left = x + 'px';
    dot.style.top = y + 'px';
    dot.style.opacity = (trailLength - index) / trailLength;

    const nextDot = yarnTrail[index + 1];
    if (nextDot) {
      x += (nextDot.offsetLeft - x) * 0.3;
      y += (nextDot.offsetTop - y) * 0.3;
    }
  });

  requestAnimationFrame(animateTrail);
}

animateTrail();
```

      ctx.lineWidth = 2;
      ctx.stroke();

      // --- Yarn Threads with Sparkling Knots & Card Sync ---
      const canvas = document.getElementById('yarn-bg');
      const ctx = canvas.getContext('2d');
      function resizeCanvas() {
        canvas.width = window.innerWidth;
        canvas.height = window.innerHeight;
      }
      window.addEventListener('resize', resizeCanvas);
      resizeCanvas();

      class YarnThread {
        constructor() {
          this.reset();
          this.knots = [];
        }
        reset() {
          this.x = Math.random() * canvas.width;
          this.y = Math.random() * canvas.height;
          this.length = 100 + Math.random() * 200;
          this.speed = 0.2 + Math.random() * 0.5;
          this.angle = Math.random() * Math.PI * 2;
          this.color = `rgba(242,140,140,${0.2 + Math.random() * 0.3})`;
          this.knots = [];
        }
        update() {
          this.x += Math.cos(this.angle) * this.speed;
          this.y += Math.sin(this.angle) * this.speed;
          this.knots.forEach((k) => (k.alpha -= 0.01));
          this.knots = this.knots.filter((k) => k.alpha > 0);
          if (
            this.x > canvas.width + 50 ||
            this.x < -50 ||
            this.y > canvas.height + 50
          )
            this.reset();
        }
        draw(ctx) {
          ctx.beginPath();
          ctx.moveTo(this.x, this.y);
          ctx.lineTo(
            this.x + Math.cos(this.angle) * this.length,
            this.y + Math.sin(this.angle) * this.length
          );
          ctx.strokeStyle = this.color;
          ctx.lineWidth = 2;
          ctx.stroke();
          this.knots.forEach((k) => {
            ctx.beginPath();
            ctx.arc(k.x, k.y, k.radius, 0, Math.PI * 2);
            ctx.fillStyle = `rgba(255,248,176,${k.alpha})`;
            ctx.fill();
          });
        }
      }

      const threads = [];
      for (let i = 0; i < 15; i++) threads.push(new YarnThread());
      function animateThreads() {
        ctx.clearRect(0, 0, canvas.width, canvas.height);
        threads.forEach((t) => {
          t.update();
          t.draw(ctx);
        });
        requestAnimationFrame(animateThreads);
      }
      animateThreads();

      function triggerKnotsNearCard(card) {
        const rect = card.getBoundingClientRect();
        const cx = rect.left + rect.width / 2;
        const cy = rect.top + rect.height / 2;
        threads.forEach((t) => {
          for (let i = 0; i < 5; i++) {
            const ox = (Math.random() - 0.5) * rect.width;
            const oy = (Math.random() - 0.5) * rect.height;
            t.knots.push({
              x: cx + ox,
              y: cy + oy,
              alpha: 1,
              radius: 2 + Math.random() * 2,
            });
          }
        });
      }

          sparkles.appendChild(s);
        }
        card.addEventListener('mouseenter', () => {

      const observer = new IntersectionObserver(
